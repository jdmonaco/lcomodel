Distance nonlinearity

This function g(d) is a nonlinearity (hyperbolic tangent) of distance d from a preferred location. It is parameterized by a radius parameter g_radius that determines the smoothness and spatial scale of the approach. The closer g_radius is to the size of the arena (80 cm) the more gradual the spatial activation.

In [4]: 
def g(d, g_radius=20, sigmoid=True):
    """ Compute a linear (or sigmoid) nonlinearity of distance """
    if sigmoid:
        return np.tanh(2 * (1 - d / g_radius))
    return 2 * (clip(g_radius - d / 2, 0, g_radius) / g_radius - 0.5)

# Plot g(d) for different values of g_radius
x = np.linspace(0,120,100) * cm
for r in array((10,15,20,30,45,60)) * cm:
    plt.plot(x / cm, g(x, g_radius=r), label='%d cm' % (r / cm))
plt.xlabel('distance (cm)')
plt.ylabel('g(distance)')
plt.legend();

LCO Neuron group

In [6]: 
eqs = """
    Phi = arccos(Ip) : radian
    Ip = tanh(2 * (1 - D / g_radius)) : 1
    D = sqrt((X - Xp)**2 + (Y - Yp)**2) : cm
    Xp : cm
    Yp : cm
    X : cm
    Y : cm
"""
LCO = NeuronGroup(N, model=eqs)

# Set the preferred locations for the LCO neurons
LCO.Xp = 2 * R_arena * (rand(N) - 0.5)
LCO.Yp = 2 * R_arena * (rand(N) - 0.5)
