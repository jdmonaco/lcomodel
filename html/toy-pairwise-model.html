def set_arena(ax=None):
    if ax is None:
        ax = gca()
    ax.add_artist(mpl.patches.Circle((0,0), radius=R_arena / cm, fill=False, ec='0.3', 
                                     lw=2, ls='dashed', zorder=-1))
    ax.axis(array([-1,1,-1,1]) * 1.2 * R_arena / cm)
    return ax

This function uniformly samples a circular disc for spatial phase offsets.

In [6]: 
def randc(N, radius):
    t = 2 * pi * rand(N)
    r = rand(N) + rand(N)
    r[r > 1] = 2 - r[r > 1]
    r *= radius
    return np.c_[r * cos(t), r * sin(t)].T

These are the equations for the BVC group of neurons:

bvc_eqs = """
    f = 0.5 + tanh(2 * (D - (1 - BVC_min) * D_arena + sigma) / sigma) / 2 : 1
    D = sqrt((X - Xp)**2 + (Y - Yp)**2) : cm
    sigma : cm
    Xp : cm # x-dimension spatial offset
    Yp : cm # y-dimension spatial offset
    X : cm
    Y : cm
"""
BVC = NeuronGroup(N_BVC, model=bvc_eqs)

BVC.Xp, BVC.Yp = randc(N_BVC, R_arena)
BVC.sigma = BVC_sigma[0] + diff(BVC_sigma) * rand(N_BVC)

def plot_phase_trajectory(cdata, ax, cmap='jet', vmin=0, vmax=pi):
    ax.scatter(X_[0] / cm, Y_[0] / cm, c=cdata, cmap=cmap, vmin=vmin, vmax=vmax, linewidths=0)
    ax.axis('equal')
    set_arena(ax)

for i in xrange(N_LCO):
    for j in xrange(N_LCO):
        if j < i:
            ax[i][j].set_axis_off()
        elif j == i:
            plot_phase_trajectory(Theta[i], ax[i][j], cmap='copper_r')
            quicktitle(ax[i][j], 'LCO %d' % i)
        else:
            plot_phase_trajectory(cdiff(Theta[i], Theta[j]), ax[i][j], cmap='hsv', vmin=-pi)
            quicktitle(ax[i][j], '%d x %d' % (i, j))
